<Attributes edit="/pages/zh-cn/components/tag.mdx">
<Attributes.Title>Tag.Props</Attributes.Title>

| 属性           | 描述                                                           | 类型                 | 推荐值                      | 默认      |
| -------------- | -------------------------------------------------------------- | -------------------- | --------------------------- | --------- |
| **variant**    | 标签的类型                                                     | `TagVariants`        | [TagVariants](#tagVariants) | `default` |
| **color**      | 标签的颜色                                                     | `TagColors`          | [TagColors](#tagcolors)     | `default` |
| **size**       | 标签的大小                                                     | `TagSizes`           | [TagSizes](#tagsizes)       | `medium`  |
| **toggleable** | 设置标签为开关                                                 | `boolean`            |                             | `false`   |
| **effect**     | 是否显示动画效果                                               | `boolean`            | -                           | `true`    |
| **dashed**     | 标签带 dashed border, 只有在 `variant` 是 `default` 的时候有效 | `boolean`            | -                           | `false`   |
| **disabled**   | 是否禁用标签                                                   | `boolean`            | -                           | `false`   |
| **onClick**    | 点击事件                                                       | `MouseEventHandler`  | -                           | -         |
| ...            | 原生属性                                                       | `SpanHTMLAttributes` | `'id', 'className', ...`    | -         |

type TagVariants = 'default' | 'solid' | 'text'

<Attributes.Title>TagColors</Attributes.Title>

```ts
type TagColors = 'default' | 'primary' | 'secondary' | 'success' | 'warning' | 'error'
```

<Attributes.Title>TagSizes</Attributes.Title>

```ts
type TagSizes = 'small' | 'medium' | 'large'
```
